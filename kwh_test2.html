sc.fit(kouhu_2)
X = sc.transform(kouhu_2)
# parametars
alpha = 0.2 # L
# Scikit LearnGraphical Lasso
model = GraphLasso(alpha=alpha,
                     max_iter=100,                     
                     verbose=True,
                     assume_centered = True)

model.fit(X)
cov_ = model.covariance_ # 
prec_ = model.precision_ #

        cor[i, j] = cov_[i, j]/np.sqrt(cov_[i, i]*cov_[j, j])
